    System.Random random = new System.Random();

    while (placedMines < mineCount)
    {
        int x = random.Next(width);
        int y = random.Next(height);

        if (!cells[x, y].IsMine)
        {
            cells[x, y].SetMine();

    }
}

public async UniTask<bool> InitTask(bool isForceInit = false)
{
    if (!isForceInit && IsInitialize) return true;
    IsInitialize = true;
    Screen.sleepTimeout = SleepTimeout.NeverSleep;

    Root = GameObject.Find(_name);
    if (Root == null) { gameObject.name = _name; Root = gameObject; }
    else if (Root != gameObject) { Destroy(gameObject); return false; }

    if (Instance == null) Instance = this;
    DontDestroyOnLoad(Root);

    // UI → Game → SFX 순서로 비동기 초기화
    await UIManager.InitTask();
    await GameManager.InitTask();
    await SFXManager.InitTask();

    Screen.sleepTimeout = SleepTimeout.SystemSetting;
    PlayBackGroundMusic();
    return true;
}
TitleSystem.cs 플레이타임 기반 칭호 지급 (비트 플래그) 
[MethodImpl(MethodImplOptions.AggressiveInlining)]
private void TryAwardTitles()
{
    if (_thresholdMinutesSorted.Length == 0) return;

    var totalMinutes = (int)(_save.totalSeconds / 60);
    var idx = _save.nextTitleIndex;

    while (idx < _thresholdMinutesSorted.Length
           && totalMinutes >= _thresholdMinutesSorted[idx])
    {
        bool newlyAwarded = true;
        if (idx < 64)
        {
            var mask = 1UL << idx;
            if ((_save.claimedBits0 & mask) != 0) newlyAwarded = false;
            else _save.claimedBits0 |= mask;   // 비트 플래그로 중복 방지
        }

        if (newlyAwarded)
        {
            var def = _definitionsSorted[idx];
            OnTitleAwarded?.Invoke(def);
            if (!def.Id.Contains("test") || Application.isEditor)
                ShowTitleToast(def);
        }
        idx++;
    }
    _save.nextTitleIndex = idx;
}

public class GameData : MonoBehaviour
{
    // 싱글톤 — 전역에서 하나의 인스턴스만 보장
    public static GameData instance;

    public GameObject Savepoint;  // 플레이어 리스폰 위치
    public float LocalTime;       // 경과 시간
    public string timeText;       // UI 표시용 문자열
    public RankSystem rank;
    public bool isStart;
    public AudioSource fireSFX;

    private void Awake() => instance = this;

    private void Update()
    {
        if (!isStart) return;
        LocalTime += Time.deltaTime;
        float sec = LocalTime % 60;
        int   min = (int)LocalTime / 60;
        timeText = $"{min:00}:{sec:00}";
    }

    public void TimeStart() => isStart = true;

    public void TimeStop()
    {
        fireSFX.Play();
        rank.RankUpdate(selectedIcon, LocalTime);
        rank.RankReset();
        rank.RankInstantiate();
        isStart = false;
    }
}
ZipLine.cs VR 양손 짚라인 인터랙션 
public class ZipLine : MonoBehaviour
{
    public XROrigin xr;
    public Transform zipPos;
    public Animator zipAnimator;
    public AudioSource ziplineSFX;

    private bool isLeftHand, isRightHand, isZip = true;

    private void Update() { if (isZip) ZipMove(); }

    private void OnTriggerEnter(Collider other)
    {
        if (other.CompareTag("LeftHand"))  isLeftHand  = true;
        if (other.CompareTag("RightHand")) isRightHand = true;
        if (other.CompareTag("Zip"))       isZip       = false;

        // 양손 모두 잡았을 때 효과음 재생
        if (isLeftHand && isRightHand) ziplineSFX.Play();
    }

    void ZipMove()
    {
        if (!isLeftHand || !isRightHand) return;
        zipAnimator.SetBool("isDown", true);
        Player.transform.position = zipPos.position; // 손잡이에 플레이어 고정
    }
}

private void OnCollisionEnter(Collision other)
{
    // 게임 시작 / 미상호작용 / 수조 내부 / 손에 없음 조건 모두 충족 시
    if (GameManagement.Instance.isStart && !hasInteracted && isAquarium && !isWall && !isHolding)
    {
        var otherSpecialty = other.gameObject.GetComponent<Specialty>();
        if (otherSpecialty != null && otherSpecialty != this && !otherSpecialty.hasInteracted)
        {
            if (specialtyData.SpecialtyName == otherSpecialty.specialtyData.SpecialtyName)
            {
                hasInteracted = true;
                otherSpecialty.hasInteracted = true;

                GameManagement.Instance.PointUp(specialtyData.SpecialtyPoint);
                GameManagement.Instance.NextSpecialtySpawn(
                    Vector3.Lerp(transform.position, other.transform.position, 0.5f),
                    specialtyData.SpecialtyName
                );
                Destroy(other.gameObject);
                Destroy(gameObject);
            }
        }
    }
    GameManagement.Instance.boop.Play();
}
